205  class late_constructed
206  {
207  public:
208  using this_type = late_constructed<T, autoConstruct, autoDestruct>;
209  using value_type = T;
210  using storage_type = eastl::aligned_storage_t<sizeof(value_type), eastl::alignment_of_v<value_type>>;
211 
212  late_constructed() EA_NOEXCEPT // In the case of the late_constructed instance being at global scope, we rely on the
213  : mStorage(), mpValue(nullptr) {} // compiler executing this constructor or placing the instance in auto-zeroed-at-startup memory.
214 
215  ~late_constructed()
216  {
217  if (autoDestruct && mpValue)
218  (*mpValue).~value_type();
219  }

223  {
224  if(!mpValue)
225  mpValue = new (&mStorage) value_type(eastl::forward<Args>(args)...);

229  { return mpValue != nullptr; }
230 
231  void destruct()
232  {

235  (*mpValue).~value_type();
236  mpValue = nullptr;
237  }
238  }

242  if(!mpValue)

245  EA_ANALYSIS_ASSUME(mpValue);

700  template <typename InputIterator, typename ForwardIterator>
701  inline ForwardIterator uninitialized_move_impl(InputIterator first, InputIterator last, ForwardIterator dest, false_type)
702  {
703  typedef typename eastl::iterator_traits<ForwardIterator>::value_type value_type;
704  ForwardIterator currentDest(dest);
705 
706  // We must run a loop over every element and move-construct it at the new location.
707  #if EASTL_EXCEPTIONS_ENABLED
708  try
709  {
710  #endif
711  for(; first != last; ++first, ++currentDest)
712  ::new((void*)eastl::addressof(*currentDest)) value_type(eastl::move(*first)); // If value_type has a move constructor then it will be used here.
713  #if EASTL_EXCEPTIONS_ENABLED
714  }
715  catch(...)
716  {
717  // We have a problem here: If an exception occurs while doing the loop below then we will
718  // have values that were moved from the source to the dest that may need to be moved back
719  // in the catch. What does the C++11 Standard say about this? And what happens if there's an
720  // exception while moving them back? We may want to trace through a conforming C++11 Standard
721  // Library to see what it does and do something similar. Given that rvalue references are
722  // objects that are going away, we may not need to move the values back, though that has the
723  // side effect of a certain kind of lost elements problem.
724  for(; dest < currentDest; ++dest)
725  (*dest).~value_type();
726  throw;
727  }
728  #endif
729 
730  return currentDest;
731  }
732  }
