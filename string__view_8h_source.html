171  static EA_CPP14_CONSTEXPR int compare(const T* pBegin1, const T* pEnd1, const T* pBegin2, const T* pEnd2)
172  {
173  const ptrdiff_t n1 = pEnd1 - pBegin1;
174  const ptrdiff_t n2 = pEnd2 - pBegin2;
175  const ptrdiff_t nMin = eastl::min_alt(n1, n2);
176  const int cmp = Compare(pBegin1, pBegin2, (size_type)nMin);

212  EA_CPP14_CONSTEXPR size_type find(basic_string_view sw, size_type pos = 0) const EA_NOEXCEPT
213  {
214  auto* pEnd = mpBegin + mnCount;
215  if (EASTL_LIKELY(((npos - sw.size()) >= pos) && (pos + sw.size()) <= mnCount))
216  {
217  const value_type* const pTemp = eastl::search(mpBegin + pos, pEnd, sw.data(), sw.data() + sw.size());
218 
219  if ((pTemp != pEnd) || (sw.size() == 0))
220  return (size_type)(pTemp - mpBegin);
221  }
222  return npos;
223  }

255  EA_CPP14_CONSTEXPR size_type rfind(const T* s, size_type pos, size_type n) const
256  {
257  // Disabled because it's not clear what values are valid for position.
258  // It is documented that npos is a valid value, though. We return npos and
259  // don't crash if postion is any invalid value.
260  //#if EASTL_ASSERT_ENABLED
261  // if(EASTL_UNLIKELY((position != npos) && (position > (size_type)(mpEnd - mpBegin))))
262  // EASTL_FAIL_MSG("basic_string::rfind -- invalid position");
263  //#endif
264 
265  // Note that a search for a zero length string starting at position = end() returns end() and not npos.
266  // Note by Paul Pedriana: I am not sure how this should behave in the case of n == 0 and position > size.
267  // The standard seems to suggest that rfind doesn't act exactly the same as find in that input position
268  // can be > size and the return value can still be other than npos. Thus, if n == 0 then you can
269  // never return npos, unlike the case with find.
270  if (EASTL_LIKELY(n <= mnCount))
271  {
272  if (EASTL_LIKELY(n))
273  {
274  const const_iterator pEnd = mpBegin + eastl::min_alt(mnCount - n, pos) + n;
275  const const_iterator pResult = CharTypeStringRSearch(mpBegin, pEnd, s, s + n);
276 
277  if (pResult != pEnd)
278  return (size_type)(pResult - mpBegin);
279  }
280  else
281  return eastl::min_alt(mnCount, pos);
282  }
283  return npos;
284  }
