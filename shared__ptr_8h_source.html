6 // This class implements the C++11 shared_ptr template. A shared_ptr is like
7 // the C++ Standard Library unique_ptr except that it allows sharing of pointers
8 // between instances via reference counting. shared_ptr objects can safely be
9 // copied and can safely be used in containers such as vector or list.
10 //
11 // Notes regarding safe usage of shared_ptr:
12 // - http://www.boost.org/doc/libs/1_53_0/libs/smart_ptr/shared_ptr.htm#ThreadSafety
13 // - If you construct a shared_ptr with a raw pointer, you cannot construct another shared_ptr
14 // with just that raw pointer. Insted you need to construct additional shared_ptrs with
15 // the originally created shared_ptr. Otherwise you will get a crash.
16 // - Usage of shared_ptr is thread-safe, but what it points to isn't automatically thread safe.
17 // Multiple shared_ptrs that refer to the same object can be used arbitrarily by multiple threads.
18 // - You can use a single shared_ptr between multiple threads in all ways except one: assigment
19 // to that shared_ptr. The following is not thread-safe, and needs to be guarded by a mutex
20 // or the shared_ptr atomic functions:
21 // shared_ptr<Foo> pFoo;
22 // // Thread 1:
23 // shared_ptr<Foo> pFoo2 = pFoo;
24 // // Thread 2:
25 // pFoo = make_shared<Foo>();
26 //
27 // Compatibility note:
28 // This version of shared_ptr updates the previous version to have full C++11
29 // compatibility. However, in order to add C++11 compatibility there needed to
30 // be a few breaking changes which may affect some users. It's likely that most
31 // or all breaking changes can be rectified by doing the same thing in a slightly
32 // different way. Here's a list of the primary signficant breaking changes:
33 // - shared_ptr now takes just one template parameter instead of three.
34 // (allocator and deleter). You now specify the allocator and deleter
35 // as part of the shared_ptr constructor at runtime.
36 // - shared_ptr has thread safety, which
37 //

181  inline ref_count_sp* ref_count_sp::lock() EA_NOEXCEPT
182  {
183  for(int32_t refCountTemp = mRefCount; refCountTemp != 0; refCountTemp = mRefCount)
184  {
185  if(Internal::atomic_compare_and_swap(&mRefCount, refCountTemp + 1, refCountTemp))
186  {
187  Internal::atomic_increment(&mWeakRefCount);
188  return this;
189  }
190  }
191 
192  return nullptr;
193  }

1213  // shared_ptr atomic access
1214  //
1215  // These functions allow shared_ptr to act like other C++11 atomic operations.
1216  // So the same way you can use atomic_load on a raw pointer, you can also
1217  // use it on a shared_ptr. This allows for transparent use of shared_ptr in
1218  // place of raw pointers (e.g. in templates). You do not need to use these
1219  // functions for regular thread-safe direct usage of shared_ptr construction
1220  // and copying, as it's intrinsically thread-safe for that already.
1221  //
1222  // That being said, the following is not thread-safe and needs to be guarded by
1223  // a mutex or the following atomic functions, as it's assigning the *same*
1224  // shared_ptr object from multiple threads as opposed to different shared_ptr
1225  // objects underlying object:
1226  // shared_ptr<Foo> pFoo;
1227  // // Thread 1:
1228  // shared_ptr<Foo> pFoo2 = pFoo;
1229  // // Thread 2:
1230  // pFoo = make_shared<Foo>();

1234  inline bool atomic_is_lock_free(const shared_ptr<T>*)
1235  {
1236  // Return true if atomic access to the provided shared_ptr instance is lock-free, false otherwise.
1237  // For this to be lock-free, we would have to be able to copy shared_ptr objects in an atomic way
1238  // as opposed to wrapping it with a mutex like we do below. Given the nature of shared_ptr, it's
1239  // probably not feasible to implement these operations without a mutex. atomic_is_lock_free exists
1240  // in the C++11 Standard because it also applies to other types such as built-in types which can
1241  // be lock-free in their access.
1242  return false;
