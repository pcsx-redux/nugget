49struct PackedVec3 {
50 Short x, y, z;
51 PackedVec3() = default;
52 explicit PackedVec3(Short x_, Short y_, Short z_) : x(x_), y(y_), z(z_) {}
53 explicit PackedVec3(const Vec3& v) {
54 x = Short(v.x);
55 y = Short(v.y);
56 z = Short(v.z);
57 }
58 operator Vec3() const {
59 Vec3 ret;
60 ret.x = FixedPoint<>(x);
61 ret.y = FixedPoint<>(y);
62 ret.z = FixedPoint<>(z);
63 return ret;
64 }

159template <Register reg, Safety safety = Safe>
160static inline void clear() {
161 if constexpr (reg < Register::R11R12) {
162 asm volatile("mtc2 $0, $%0" ::"i"(static_cast<uint32_t>(reg)));
163 } else if constexpr (reg >= Register::R11R12) {
164 asm volatile("ctc2 $0, $%0" ::"i"(static_cast<uint32_t>(reg) - 32));
165 }
166 if constexpr (safety == Safe) {
167 asm volatile("nop; nop");
168 }
169}

178template <Register reg, Safety safety = Safe>
179static inline void write(uint32_t value) {
180 if (__builtin_constant_p(value) && (value == 0)) {
181 clear<reg, safety>();
182 } else {
183 if constexpr (reg < Register::R11R12) {
184 asm volatile("mtc2 %0, $%1" ::"r"(value), "i"(static_cast<uint32_t>(reg)));
185 } else if constexpr (reg >= Register::R11R12) {
186 asm volatile("ctc2 %0, $%1" ::"r"(value), "i"(static_cast<uint32_t>(reg) - 32));
187 }
188 if constexpr (safety == Safe) {
189 asm volatile("nop; nop");
190 }
191 }
192}

303template <Register reg, Safety safety = Safe>
304static inline void write(const uint32_t* ptr) {
305 static_assert(reg < Register::R11R12, "Unable to write to register from memory directly");
306 if constexpr (reg < Register::R11R12) {
307 asm volatile("lwc2 $%1, 0(%0)" ::"r"(ptr), "i"(static_cast<uint32_t>(reg)));
308 }
309
310 if constexpr (safety == Safe) {
311 asm volatile("nop; nop");
312 }
313}
314
322template <Register reg, Safety safety = Safe>
323static inline uint32_t read() {
324 uint32_t value;
325 if constexpr (reg < Register::R11R12) {
326 if constexpr (safety == Safe) {
327 asm volatile("mfc2 %0, $%1; nop" : "=r"(value) : "i"(static_cast<uint32_t>(reg)));
328 } else if constexpr (safety == Unsafe) {
329 asm volatile("mfc2 %0, $%1" : "=r"(value) : "i"(static_cast<uint32_t>(reg)));
330 }
331 } else if constexpr (reg >= Register::R11R12) {
332 if constexpr (safety == Safe) {
333 asm volatile("cfc2 %0, $%1; nop" : "=r"(value) : "i"(static_cast<uint32_t>(reg) - 32));
334 } else if constexpr (safety == Unsafe) {
335 asm volatile("cfc2 %0, $%1" : "=r"(value) : "i"(static_cast<uint32_t>(reg) - 32));
336 }
337 }
338 return value;
339}

348static inline void read(uint32_t* ptr) {
349 static_assert(reg < Register::R11R12, "Unable to read from register to memory directly");
350 if constexpr (reg < Register::R11R12) {
351 asm volatile("swc2 $%1, 0(%0)" ::"r"(ptr), "i"(static_cast<uint32_t>(reg)));
