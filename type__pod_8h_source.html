215  // has_trivial_copy
216  //
217  // has_trivial_copy<T>::value == true if and only if T is a class or
218  // struct that has a trivial copy constructor. A copy constructor is
219  // trivial if
220  // - it is implicitly defined by the compiler, and
221  // - is_polymorphic<T>::value == false, and
222  // - T has no virtual base classes, and
223  // - for every direct base class of T, has_trivial_copy<B>::value == true,
224  // where B is the type of the base class, and
225  // - for every nonstatic data member of T that has class type or array
226  // of class type, has_trivial_copy<M>::value == true, where M is the
227  // type of the data member
228  //
229  // has_trivial_copy may only be applied to complete types.
230  //
231  // Another way of looking at this is:
232  // A copy constructor for class X is trivial if it is implicitly
233  // declared and if all the following are true:
234  // - Class X has no virtual functions (10.3) and no virtual base classes (10.1).
235  // - Each direct base class of X has a trivial copy constructor.
236  // - For all the nonstatic data members of X that are of class type
237  // (or array thereof), each such class type has a trivial copy constructor;
238  // otherwise the copy constructor is nontrivial.
239  //
240  // Without help from the compiler or user, has_trivial_copy will not report
241  // that a class or struct has a trivial copy constructor. The user can
242  // use EASTL_DECLARE_TRIVIAL_COPY to help the compiler.

1264  // is_assignable
1265  //
1266  // The expression declval<T>() = declval<U>() is well-formed when treated as an unevaluated operand.
1267  // Access checking is performed as if in a context unrelated to T and U. Only the validity of
1268  // the immediate context of the assignment expression is considered. The compilation of the expression
1269  // can result in side effects such as the instantiation of class template specializations and function
1270  // template specializations, the generation of implicitly-defined functions, and so on. Such side
1271  // effects are not in the "immediate context" and can result in the program being ill-formed.
1272  //
1273  // Note:
1274  // This type trait has a misleading and counter-intuitive name. It does not indicate whether an instance
1275  // of U can be assigned to an instance of T (e.g. t = u). Instead it indicates whether the assignment can be
1276  // done after adding rvalue references to both, as in add_rvalue_reference<T>::type = add_rvalue_reference<U>::type.
1277  // A counterintuitive result of this is that is_assignable<int, int>::value == false. The is_copy_assignable
1278  // trait indicates if a type can be assigned to its own type, though there isn't a standard C++ way to tell
1279  // if an arbitrary type is assignable to another type.
1280  // http://stackoverflow.com/questions/19920213/why-is-stdis-assignable-counter-intuitive
1281  //
1282  // Note:
1283  // A true is_assignable value doesn't guarantee that the expression is compile-able, the compiler checks
1284  // only that the assignment matches before compilation. In particular, if you have templated operator=
1285  // for a class, the compiler will always say is_assignable is true, regardless of what's being tested
1286  // on the right hand side of the expression. It may actually turn out during compilation that the
1287  // templated operator= fails to compile because in practice it doesn't accept every possible type for
1288  // the right hand side of the expression.
1289  //
1290  // Expected results:
1291  // is_assignable<void, void>::value == false
1292  // is_assignable<int&, int>::value == true
1293  // is_assignable<int, int>::value == false
1294  // is_assignable<int, int&>::value == false
1295  // is_assignable<bool, bool>::value == false
1296  // is_assignable<int, float>::value == false
1297  // is_assignable<int[], int[]>::value == false
1298  // is_assignable<char*, int*>::value == false
1299  // is_assignable<char*, const char*>::value == false
1300  // is_assignable<const char*, char*>::value == false
1301  // is_assignable<PodA, PodB*>::value == false
1302  // is_assignable<Assignable, Assignable>::value == true
1303  // is_assignable<Assignable, Unrelated>::value == false
1304  //
1305  // Note:
1306  // Our implementation here yields different results than does the std::is_assignable from Dinkumware-based Standard
1307  // Libraries, but yields similar results to the std::is_assignable from GCC's libstdc++ and clang's libc++. It may
1308  // possibly be that the Dinkumware results are intentionally different for some practical purpose or because they
1309  // represent the spirit or the Standard but not the letter of the Standard.
1310  //
1312  #define EASTL_TYPE_TRAIT_is_assignable_CONFORMANCE 1
