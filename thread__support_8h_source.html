98  inline int32_t atomic_increment(int32_t* p32) EA_NOEXCEPT
99  {
100  #if defined(__clang__) || (defined(EA_COMPILER_GNUC) && (EA_COMPILER_VERSION >= 4003))
101  return __sync_add_and_fetch(p32, 1);
102  #elif defined(EA_COMPILER_MSVC)
103  static_assert(sizeof(long) == sizeof(int32_t), "unexpected size");
104  return _InterlockedIncrement((volatile long*)p32);
105  #elif defined(EA_COMPILER_GNUC)
106  int32_t result;
107  __asm__ __volatile__ ("lock; xaddl %0, %1"
108  : "=r" (result), "=m" (*p32)
109  : "0" (1), "m" (*p32)
110  : "memory"
111  );
112  return result + 1;
113  #else
114  EASTL_FAIL_MSG("EASTL thread safety is not implemented yet. See EAThread for how to do this for the given platform.");
115  return ++*p32;
116  #endif
117  }

121  inline int32_t atomic_decrement(int32_t* p32) EA_NOEXCEPT
122  {
123  #if defined(__clang__) || (defined(EA_COMPILER_GNUC) && (EA_COMPILER_VERSION >= 4003))
124  return __sync_add_and_fetch(p32, -1);
125  #elif defined(EA_COMPILER_MSVC)
126  return _InterlockedDecrement((volatile long*)p32); // volatile long cast is OK because int32_t == long on Microsoft platforms.
127  #elif defined(EA_COMPILER_GNUC)
128  int32_t result;
129  __asm__ __volatile__ ("lock; xaddl %0, %1"
130  : "=r" (result), "=m" (*p32)
131  : "0" (-1), "m" (*p32)
132  : "memory"
133  );
134  return result - 1;
135  #else
136  EASTL_FAIL_MSG("EASTL thread safety is not implemented yet. See EAThread for how to do this for the given platform.");
137  return --*p32;
138  #endif

148  inline bool atomic_compare_and_swap(int32_t* p32, int32_t newValue, int32_t condition)
149  {
150  #if defined(__clang__) || (defined(EA_COMPILER_GNUC) && (EA_COMPILER_VERSION >= 4003))
151  return __sync_bool_compare_and_swap(p32, condition, newValue);
152  #elif defined(EA_COMPILER_MSVC)
153  return ((int32_t)_InterlockedCompareExchange((volatile long*)p32, (long)newValue, (long)condition) == condition);
154  #elif defined(EA_COMPILER_GNUC)
155  // GCC Inline ASM Constraints
156  // r <--> Any general purpose register
157  // a <--> The a register.
158  // 1 <--> The constraint '1' for operand 2 says that it must occupy the same location as operand 1.
159  // =a <--> output registers
160  // =r <--> output registers
161 
162  int32_t result;
163  __asm__ __volatile__(
164  "lock; cmpxchgl %3, (%1) \n" // Test *p32 against EAX, if same, then *p32 = newValue
165  : "=a" (result), "=r" (p32) // outputs
166  : "a" (condition), "r" (newValue), "1" (p32) // inputs
167  : "memory" // clobbered
168  );
169  return result == condition;
170  #else
171  EASTL_FAIL_MSG("EASTL thread safety is not implemented yet. See EAThread for how to do this for the given platform.");
172  if(*p32 == condition)
173  {
174  *p32 = newValue;
175  return true;
176  }
177  return false;
178  #endif
179  }
