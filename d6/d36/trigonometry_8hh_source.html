84 constexpr FixedPoint<precisionBits> cos(Angle a) const {
85 using namespace trig_literals;

88 t %= (2.0_pi).value;
89 a.value = t;
90 int32_t r;
91
92 if (a < 0.5_pi) {
93 r = table[t];
94 } else if (a < 1.0_pi) {
95 r = -table[(1.0_pi).value - 1 - t];
96 } else if (a < 1.5_pi) {
97 r = -table[t - (1.0_pi).value];
98 } else {
99 r = table[(2.0_pi).value - 1 - t];
100 }
101
102 FixedPoint<precisionBits> ret;
103 ret.value = r;
104 return ret;
105 }
