1026  template <typename K, typename V, typename C, typename A, typename E, bool bM, bool bU>
1027  void rbtree<K, V, C, A, E, bM, bU>::swap(this_type& x)
1028  {
1029  #if EASTL_RBTREE_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR
1030  if(mAllocator == x.mAllocator) // If allocators are equivalent...
1031  #endif
1032  {
1033  // Most of our members can be exchaged by a basic swap:
1034  // We leave mAllocator as-is.
1035  eastl::swap(mnSize, x.mnSize);
1036  eastl::swap(get_compare(), x.get_compare());
1037  #if !EASTL_RBTREE_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR
1038  eastl::swap(mAllocator, x.mAllocator);
1039  #endif
1040 
1041 
1042  // However, because our anchor node is a part of our class instance and not
1043  // dynamically allocated, we can't do a swap of it but must do a more elaborate
1044  // procedure. This is the downside to having the mAnchor be like this, but
1045  // otherwise we consider it a good idea to avoid allocating memory for a
1046  // nominal container instance.
1047 
1048  // We optimize for the expected most common case: both pointers being non-null.
1049  if(mAnchor.mpNodeParent && x.mAnchor.mpNodeParent) // If both pointers are non-null...
1050  {
1051  eastl::swap(mAnchor.mpNodeRight, x.mAnchor.mpNodeRight);
1052  eastl::swap(mAnchor.mpNodeLeft, x.mAnchor.mpNodeLeft);
1053  eastl::swap(mAnchor.mpNodeParent, x.mAnchor.mpNodeParent);
1054 
1055  // We need to fix up the anchors to point to themselves (we can't just swap them).
1056  mAnchor.mpNodeParent->mpNodeParent = &mAnchor;
1057  x.mAnchor.mpNodeParent->mpNodeParent = &x.mAnchor;
1058  }
1059  else if(mAnchor.mpNodeParent)
1060  {
1061  x.mAnchor.mpNodeRight = mAnchor.mpNodeRight;
1062  x.mAnchor.mpNodeLeft = mAnchor.mpNodeLeft;
1063  x.mAnchor.mpNodeParent = mAnchor.mpNodeParent;
1064  x.mAnchor.mpNodeParent->mpNodeParent = &x.mAnchor;
1065 
1066  // We need to fix up our anchor to point it itself (we can't have it swap with x).
1067  mAnchor.mpNodeRight = &mAnchor;
1068  mAnchor.mpNodeLeft = &mAnchor;
1069  mAnchor.mpNodeParent = NULL;
1070  }
1071  else if(x.mAnchor.mpNodeParent)
1072  {
1073  mAnchor.mpNodeRight = x.mAnchor.mpNodeRight;
1074  mAnchor.mpNodeLeft = x.mAnchor.mpNodeLeft;
1075  mAnchor.mpNodeParent = x.mAnchor.mpNodeParent;
1076  mAnchor.mpNodeParent->mpNodeParent = &mAnchor;
1077 
1078  // We need to fix up x's anchor to point it itself (we can't have it swap with us).
1079  x.mAnchor.mpNodeRight = &x.mAnchor;
1080  x.mAnchor.mpNodeLeft = &x.mAnchor;
1081  x.mAnchor.mpNodeParent = NULL;
1082  } // Else both are NULL and there is nothing to do.
1083  }
1084  #if EASTL_RBTREE_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR
1085  else
1086  {
1087  const this_type temp(*this); // Can't call eastl::swap because that would
1088  *this = x; // itself call this member swap function.
1089  x = temp;
1090  }
1091  #endif
1092  }
