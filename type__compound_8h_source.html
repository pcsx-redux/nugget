30  //
31  // For a given array type T[N], extent<T[N]>::value == N.

268  // is_convertible
269  //
270  // Given two (possible identical) types From and To, is_convertible<From, To>::value == true
271  // if and only if an lvalue of type From can be implicitly converted to type To,
272  // or is_void<To>::value == true
273  //
274  // An instance of the type predicate holds true if the expression To to = from;, where from is an object of type From, is well-formed.
275  //
276  // is_convertible may only be applied to complete types.
277  // Type To may not be an abstract type.
278  // If the conversion is ambiguous, the program is ill-formed.
279  // If either or both of From and To are class types, and the conversion would invoke
280  // non-public member functions of either From or To (such as a private constructor of To,
281  // or a private conversion operator of From), the program is ill-formed.
282  //
283  // Note that without compiler help, both is_convertible and is_base
284  // can produce compiler errors if the conversion is ambiguous.
285  // Example:
286  // struct A {};
287  // struct B : A {};
288  // struct C : A {};
289  // struct D : B, C {};
290  // is_convertible<D*, A*>::value; // Generates compiler error.
292 
293  #if EASTL_COMPILER_INTRINSIC_TYPE_TRAITS_AVAILABLE && (defined(_MSC_VER) || (defined(__clang__) && EA_COMPILER_HAS_FEATURE(is_convertible_to)))
294  #define EASTL_TYPE_TRAIT_is_convertible_CONFORMANCE 1 // is_convertible is conforming.
295 
296  // Problem: VC++ reports that int is convertible to short, yet if you construct a short from an int then VC++ generates a warning:
297  // warning C4242: 'initializing' : conversion from 'int' to 'short', possible loss of data. We can deal with this by making
298  // is_convertible be false for conversions that could result in loss of data. Or we could make another trait called is_lossless_convertible
299  // and use that appropriately in our code. Or we could put the onus on the user to work around such warnings.
300  template <typename From, typename To>
301  struct is_convertible : public integral_constant<bool, __is_convertible_to(From, To)>{};
302 
303  #else
304  #define EASTL_TYPE_TRAIT_is_convertible_CONFORMANCE 1
