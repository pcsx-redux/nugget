64template <unsigned precisionBits = 12, std::integral T = int32_t>
65class FixedPoint {
66 using signedUpType = std::conditional<sizeof(T) == 4, int64_t, int32_t>::type;
67 using unsignedUpType = std::conditional<sizeof(T) == 4, uint64_t, uint32_t>::type;
68 using upType = std::conditional<std::is_signed<T>::value, signedUpType, unsignedUpType>::type;

85 static constexpr unsigned scale = 1 << precisionBits;

91 explicit constexpr FixedPoint(T integer, T fraction) : value(integer * scale + fraction) {
92 static_assert(sizeof(T) == 4 || sizeof(T) == 2);
93 static_assert(precisionBits > 0);
94 }

106 consteval FixedPoint(long double ld) {
107 static_assert(std::is_signed<T>::value || (ld >= 0));
108 bool negative = ld < 0;
109 T integer = negative ? -ld : ld;
110 T fraction = ld * scale - integer * scale + (negative ? -0.5 : 0.5);
111 value = integer * scale + fraction;
112 }

114 constexpr FixedPoint() : value(0) {}

126 template <unsigned otherPrecisionBits = 12, std::integral U = int32_t>
127 explicit FixedPoint(FixedPoint<otherPrecisionBits, U> other) {
128 if constexpr (precisionBits == otherPrecisionBits) {
129 value = T(other.value);
130 } else if constexpr (precisionBits > otherPrecisionBits) {
131 value = T(other.value << (precisionBits - otherPrecisionBits));
132 } else if constexpr (precisionBits < otherPrecisionBits) {
133 value = T(other.value >> (otherPrecisionBits - precisionBits));
134 }
135 }

153 template <size_t factor = 1>
154 constexpr T integer() const {
155 if constexpr (std::is_signed<T>::value) {
156 if (value < 0) {
157 return (value - scale / (2 * factor)) / (scale / factor);
158 }
159 }
160 return (value + scale / (2 * factor)) / (scale / factor);
161 }
162
163 template <std::integral U>
164 constexpr U integer() const {
165 if constexpr (std::is_signed<T>::value) {
166 if (value < 0) {
167 return U((value - scale / 2) / scale);
168 }
169 }
170 return U((value + scale / 2) / scale);
171 }

400template <unsigned precisionBits = 12, std::integral T = int32_t, std::integral U = int32_t>
401constexpr FixedPoint<precisionBits, T> operator/(U a, FixedPoint<precisionBits, T> b) {
402 FixedPoint<precisionBits, T> ret;
403 if constexpr (sizeof(T) == 4) {
404 if constexpr (std::is_signed<T>::value || std::is_signed<U>::value) {
405 ret.set(FixedPointInternals::iDiv(a, b.get()));
406 } else if constexpr (!std::is_signed<T>::value && !std::is_signed<U>::value) {
407 ret.set(FixedPointInternals::dDiv(a, b.get()));
408 }
409 } else if constexpr (sizeof(T) == 2) {
410 ret.set(a * FixedPoint<precisionBits, T>::scale / b.get());
411 }
412 return ret;
413}
