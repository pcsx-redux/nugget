45enum {

47 SIO_CTRL_DTR = (1 << 1), // Data Terminal Ready, aka Select (output)

60enum {

64 SIO_STAT_PE = (1 << 3), // Parity Error

66 SIO_STAT_FE = (1 << 5), // Framing Error
67 SIO_STAT_SYNDET = (1 << 6), // Sync Detect
