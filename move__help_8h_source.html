69  // forward
70  //
71  // forwards the argument to another function exactly as it was passed to the calling function.
72  // Not to be confused with move, this is specifically for echoing templated argument types
73  // to another function. move is specifically about making a type be an rvalue reference (i.e. movable) type.
74  //
75  // Example usage:
76  // template <class T>
77  // void WrapperFunction(T&& arg)
78  // { foo(eastl::forward<T>(arg)); }
79  //
80  // template <class... Args>
81  // void WrapperFunction(Args&&... args)
82  // { foo(eastl::forward<Args>(args)...); }
83  //
84  // See the C++ Standard, section 20.2.3
85  // http://en.cppreference.com/w/cpp/utility/forward
86  //
87  template <typename T>
88  EA_CPP14_CONSTEXPR T&& forward(typename eastl::remove_reference<T>::type& x) EA_NOEXCEPT
89  {
90  return static_cast<T&&>(x);
91  }

102  // move
103  //
104  // move obtains an rvalue reference to its argument and converts it to an xvalue.
105  // Returns, by definition: static_cast<typename remove_reference<T>::type&&>(t).
106  // The primary use of this is to pass a move'd type to a function which takes T&&,
107  // and thus select that function instead of (e.g.) a function which takes T or T&.
108  // See the C++ Standard, section 20.2.3
109  // http://en.cppreference.com/w/cpp/utility/move
110  //
111  template <typename T>
112  EA_CPP14_CONSTEXPR typename eastl::remove_reference<T>::type&&
113  move(T&& x) EA_NOEXCEPT
114  {
115  return static_cast<typename eastl::remove_reference<T>::type&&>(x);
116  }

119  // move_if_noexcept
120  //
121  // Returns T&& if move-constructing T throws no exceptions. Instead returns const T& if
122  // move-constructing T throws exceptions or has no accessible copy constructor.
123  // The purpose of this is to use automatically use copy construction instead of move
124  // construction when the move may possible throw an exception.
125  // See the C++ Standard, section 20.2.3
126  // http://en.cppreference.com/w/cpp/utility/move_if_noexcept
127  //
128  #if EASTL_EXCEPTIONS_ENABLED
129  template <typename T>
130  EA_CPP14_CONSTEXPR typename eastl::conditional<!eastl::is_nothrow_move_constructible<T>::value &&
131  eastl::is_copy_constructible<T>::value, const T&, T&&>::type
132  move_if_noexcept(T& x) EA_NOEXCEPT
133  {
134  return eastl::move(x);
135  }
136  #else
137  template <typename T>
138  EA_CPP14_CONSTEXPR T&&
139  move_if_noexcept(T& x) EA_NOEXCEPT
