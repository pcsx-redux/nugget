543  // aligned_storage
544  //
545  // The aligned_storage transformation trait provides a type that is
546  // suitably aligned to store an object whose size is does not exceed length
547  // and whose alignment is a divisor of alignment. When using aligned_storage,
548  // length must be non-zero, and alignment must >= alignment_of<T>::value
549  // for some type T. We require the alignment value to be a power-of-two.
550  //
551  // GCC versions prior to 4.4 don't properly support this with stack-based
552  // variables. The EABase EA_ALIGN_MAX_AUTOMATIC define identifies the
553  // extent to which stack (automatic) variables can be aligned for the
554  // given compiler/platform combination.
555  //
556  // Example usage:
557  // aligned_storage<sizeof(Widget), alignment_of(Widget)>::type widget;
558  // Widget* pWidget = new(&widget) Widget;
559  //
560  // aligned_storage<sizeof(Widget), 64>::type widgetAlignedTo64;
561  // Widget* pWidget = new(&widgetAlignedTo64) Widget;
562  //
563  // aligned_storage<sizeof(Widget), alignment_of(Widget)>::type widgetArray[37];
564  // Widget* pWidgetArray = new(widgetArray) Widget[37];
566 
567  #define EASTL_TYPE_TRAIT_aligned_storage_CONFORMANCE 1 // aligned_storage is conforming.

635  // aligned_union
636  //
637  // The member typedef type shall be a POD type suitable for use as
638  // uninitialized storage for any object whose type is listed in Types;
639  // its size shall be at least Len. The static member alignment_value
640  // shall be an integral constant of type std::size_t whose value is
641  // the strictest alignment of all types listed in Types.
642  // Note that the resulting type is not a C/C++ union, but simply memory
643  // block (of pod type) that can be used to placement-new an actual
644  // C/C++ union of the types. The actual union you declare can be a non-POD union.
645  //
646  // Example usage:
647  // union MyUnion {
648  // char c;
649  // int i;
650  // float f;
651  //
652  // MyUnion(float fValue) : f(fValue) {}
653  // };

710  // union_cast
711  //
712  // Safely converts between unrelated types that have a binary equivalency.
713  // This appoach is required by strictly conforming C++ compilers because
714  // directly using a C or C++ cast between unrelated types is fraught with
715  // the possibility of undefined runtime behavior due to type aliasing.
716  // The Source and Dest types must be POD types due to the use of a union
717  // in C++ versions prior to C++11. C++11 relaxes the definition of a POD
718  // such that it allows a classes with trivial default constructors whereas
719  // previous versions did not, so beware of this when writing portable code.
720  //
721  // Example usage:
722  // float f32 = 1.234f;
723  // uint32_t n32 = union_cast<uint32_t>(f32);
724  //
725  // Example possible mis-usage:
726  // The following is valid only if you are aliasing the pointer value and
727  // not what it points to. Most of the time the user intends the latter,
728  // which isn't strictly possible.
729  // Widget* pWidget = CreateWidget();
730  // Foo* pFoo = union_cast<Foo*>(pWidget);
732 
733  template <typename DestType, typename SourceType>
734  DestType union_cast(SourceType sourceValue)
735  {
736  EASTL_CT_ASSERT((sizeof(DestType) == sizeof(SourceType)) &&
737  (EA_ALIGN_OF(DestType) == EA_ALIGN_OF(SourceType))); // To support differening alignments, we would need to use a memcpy-based solution or find a way to make the two union members align with each other.
738  //EASTL_CT_ASSERT(is_pod<DestType>::value && is_pod<SourceType>::value); // Disabled because we don't want to restrict what the user can do, as some compiler's definitions of is_pod aren't up to C++11 Standards.
739  //EASTL_CT_ASSERT(!is_pointer<DestType>::value && !is_pointer<SourceType>::value); // Disabled because it's valid to alias pointers as long as you are aliasong the pointer value and not what it points to.
740 
741  union {
742  SourceType sourceValue;
743  DestType destValue;
744  } u;
745  u.sourceValue = sourceValue;
746 
747  return u.destValue;
748  }
