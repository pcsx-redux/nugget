133  // unique_type_list
134  //
135  // unique_type_list is a meta-function which takes a parameter pack as its
136  // argument, and returns a type_list with duplicate types removed, like so:
137  //
138  // unique_type_list<int, int, string>::type; // type = type_list<int, string>
139  // unique_type_list<int, string, string>::type; // type = type_list<int, string>
140  //
141  // To use unique_type_list, specialize a variadic class template for a single
142  // type parameter, which is type_list<Ts...>:
143  //
144  // template <typename... Ts> struct foo {};
145  // template <typename... Ts> struct foo<type_list<Ts...>> {};
146  //
147  // Then instantiate the template with unique_type_list_t<Ts...> as its parameter:
148  //
149  // template <typename... Ts> struct bar : public foo<unique_type_list_t<Ts...>> {}
150  //
151  // See overload_set below for examples.
152  template <typename T, typename... Ts>
153  struct unique_type_list : public unique_type_list<Ts...>
154  {
155  template <typename... Args>
156  static enable_if_t<!disjunction_v<is_same<T, Args>...>, type_list<T, Args...>>
157  types(type_list<Args...>);
158 
159  template <typename... Args>
160  static enable_if_t<disjunction_v<is_same<T, Args>...>, type_list<Args...>>
161  types(type_list<Args...>);
162 
163  typedef decltype(types(declval<typename unique_type_list<Ts...>::type>())) type;
164  };
165 
166  template <typename T>
167  struct unique_type_list<T>
168  {
169  using type = type_list<T>;
170  };
171 
172  template <typename... Ts>
173  using unique_type_list_t = typename unique_type_list<Ts...>::type;
