6 // This file implements vector_multimap. It acts much like std::multimap, except
7 // its underlying representation is a random access container such as vector.
8 // These containers are sometimes also known as "sorted vectors."
9 // vector_maps have an advantage over conventional maps in that their memory
10 // is contiguous and node-less. The result is that lookups are faster, more
11 // cache friendly (which potentially more so benefits speed), and the container
12 // uses less memory. The downside is that inserting new items into the container
13 // is slower if they are inserted in random order instead of in sorted order.
14 // This tradeoff is well-worth it for many cases. Note that vector_multimap allows
15 // you to use a deque or other random access container which may perform
16 // better for you than vector.
17 //
18 // Note that with vector_set, vector_multiset, vector_map, vector_multimap
19 // that the modification of the container potentially invalidates all
20 // existing iterators into the container, unlike what happens with conventional
21 // sets and maps.

516  template <typename K, typename T, typename C, typename A, typename RAC>
517  inline typename vector_multimap<K, T, C, A, RAC>::iterator
518  vector_multimap<K, T, C, A, RAC>::insert(const_iterator position, const value_type& value)
519  {
520  // We assume that the user knows what he is doing and has supplied us with
521  // a position that is right where value should be inserted (put in front of).
522  // We do a test to see if the position is correct. If so then we insert,
523  // if not then we ignore the input position. However,
524 
525  if((position == end()) || !mValueCompare(*position, value)) // If value is <= the element at position...
526  {
527  if((position == begin()) || !mValueCompare(value, *(position - 1))) // If value is >= the element before position...
528  return base_type::insert(position, value);
529  }
530 
531  // In this case we have an incorrect position. We fall back to the regular insert function.
532  return insert(value);
533  }
