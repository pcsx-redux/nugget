83 enum PadType : uint8_t {

85 NegCon = 0x23, // (steering twist/wheel/paddle)

88 AnalogStick = 0x53, // (or analog pad in "flight mode"; LED=Green)
89 NamcoLightgun = 0x63, // (Cinch-type)
90 AnalogPad = 0x73, // (in normal analog mode; LED=Red)
91 Multitap = 0x80, // (multiplayer adaptor) (when activated)
92 Jogcon = 0xe3, // (steering dial)
93 ConfigMode = 0xf3, // (when in config mode; see rumble command 43h)
94 None = 0xff // (no controller connected, pins floating High-Z)
95 };
