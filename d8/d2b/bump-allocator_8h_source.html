54template <size_t N>
55class BumpAllocator {
56 public:
57 template <Primitive P, typename... Args>
58 Fragments::SimpleFragment<P> &allocateFragment(Args &&...args) {
59 static constexpr size_t size = sizeof(Fragments::SimpleFragment<P>);
60 psyqo::Kernel::assert(remaining() >= size, "BumpAllocator: Out of memory");
61 uint8_t *ptr = m_current;
62 m_current += size;
63 return *new (ptr) Fragments::SimpleFragment<P>(eastl::forward<Args>(args)...);
64 }
65 template <typename T, typename... Args>
66 T &allocate(Args &&...args) {
67 size_t size = sizeof(T);
68 uint8_t *ptr = m_current;
69 if constexpr (alignof(T) > 1) {
70 static constexpr size_t a = alignof(T) - 1;
71 auto alignedptr = reinterpret_cast<uint8_t *>((reinterpret_cast<uintptr_t>(ptr) + a) & ~a);
72 size += alignedptr - ptr;
73 ptr = alignedptr;
74 }
75 psyqo::Kernel::assert(remaining() >= size, "BumpAllocator: Out of memory");
76 m_current += size;
77 return *new (ptr) T(eastl::forward<Args>(args)...);
78 }
79 void reset() { m_current = m_memory; }
80 size_t remaining() const { return N - (m_current - m_memory); }
81 size_t used() const { return m_current - m_memory; }
82
83 private:
84 uint8_t m_memory[N] __attribute__((aligned(4)));
85 uint8_t *m_current = m_memory;
86};
